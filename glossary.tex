\newglossaryentry{gls:elempoly}{
	name={Elementary polynomials},
	description={
		All symmetric polynomials in $x_{1},\ldots,x_{n}$ can be written as a sum of \textit{elementary polynomials}. Elementary polynomials $E_{k}$ are defined as
			\begin{align*}
				E_{0}(x_{1},\ldots,x_{n})
					& = 1\\
				E_{1}(x_{1},\ldots,x_{n})
					& = x_{1}+\ldots+x_{n}\\
				E_{2}(x_{1},\ldots,x_{n})
					& = x_{1}x_{2}+x_{1}x_{3}+\ldots+x_{n-1}x_{n}\\
					& = \sum_{1\leq i<j\leq n}x_{i}x_{j}\\
					& \vdots\\
				E_{k}(x_{1},\ldots,x_{n})
					& = \sum_{1\leq i_{1}<\ldots<i_{n}\leq n}x_{i_{1}}\cdots x_{i_{n}}
			\end{align*}
		If $k>n$, then $E_{k}(x_{1},\ldots,x_{n})=0$. A special case which is very useful for us is
			\begin{align*}
				E_{1}(a,b,c)
					& = a+b+c\\
				E_{2}(a,b,c)
					& = ab+bc+ca\\
				E_{3}(a,b,c)
					& = abc
			\end{align*}
		We can show that any symmetric polynomial can be written as a sum of elementary polynomials using induction. The main idea behind this is to show that
			\begin{align*}
				\sigma_{k}
					& =x_{1}^{k}+\ldots+x_{n}^{k}
			\end{align*}
		can be represented in terms of $E_{0},\ldots,E_{k}$. For $k\in\{0,1\}$, the proof is obvious.
	}
}
